26 /** Base trait for substitution mutation
27   *
28   * Can implicitly be converted to the appropriate `scala.meta.Tree` by importing

30   *
31   * @tparam T
32   *   Has to be a subtype of Tree. This is so that the tree value and unapply methods return the appropriate type. E.G.

35 trait SubstitutionMutation[T <: Tree] extends Mutation[T] with NoInvalidPlacement[T] {

38   override def unapply(arg: T): Option[T] = {

40       .filter(_.isEqual(tree))
