32     val message = StartTestRun(mutant.id, testNames)
33     testProcess.sendMessage(message).map {
34       case TestsSuccessful(testsCompleted) =>
35         mutant.toMutantResult(MutantStatus.Survived, testsCompleted = testsCompleted.some)

37         mutant.toMutantResult(MutantStatus.Killed, testsCompleted = testsCompleted.some)
38       case ErrorDuringTestRun(msg) => mutant.toMutantResult(MutantStatus.Killed, description = msg.some)
39       case _                       => mutant.toMutantResult(MutantStatus.RuntimeError)
40     }

43   /** Initial test-run is done twice. This allows us to collect coverage data while filtering out 'static' mutants.
44     * Mutants are considered static if they are initialized only once. This means the value cannot be changed using
45     * mutation switching. For example, a `val a = 2` inside an `object` is considered static.
46     *
47     * In the first initial test-run, coverage data is collected. When running the second time any static mutants will
48     * not have coverage because their code will not be executed a second time, so we can filter those out.
49     *
50     * @see
51     *   https://github.com/stryker-mutator/stryker4s/pull/565#issuecomment-688438699
52     */
53   override def initialTestRun(): IO[InitialTestRunResult] = {
54     val initialTestRun = testProcess.sendMessage(StartInitialTestRun())
55 
56     initialTestRun.map2(initialTestRun) {
57       case (firstRun: CoverageTestRunResult, secondRun: CoverageTestRunResult) =>
58         val averageDuration =
59           FiniteDuration((firstRun.durationNanos + secondRun.durationNanos) / 2, TimeUnit.NANOSECONDS)
60 
61         InitialTestRunCoverageReport(
62           firstRun.isSuccessful && secondRun.isSuccessful,
63           CoverageReport(firstRun.coverageTestNameMap.get),
64           CoverageReport(secondRun.coverageTestNameMap.get),
65           averageDuration
66         )
67       case x => throw new MatchError(x)
68     }
69   }
